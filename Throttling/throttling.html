// 📏 Resize Throttling
let resizeCount = 0;
let throttledResizeCount = 0;
window.addEventListener('resize', () => {
resizeCount++;
console.log('Resize:', resizeCount);
});
window.addEventListener('resize', throttle(() => {
throttledResizeCount++;
console.log('Throttled Resize:', throttledResizeCount);
}, 500));
