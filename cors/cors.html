üìã Important CORS Headers

Header Purpose Example Simple Explanation
Access-Control-Allow-Origin Specifies which domains are allowed to access resources * or http://localhost:3000 "Allow all domains" or "Allow specific domain"
Access-Control-Allow-Methods Lists allowed HTTP methods GET, POST, PUT, DELETE "These actions are permitted"
Access-Control-Allow-Headers Defines allowed request headers Content-Type, Authorization "These headers can be sent"
Access-Control-Allow-Credentials Enables sending cookies or credentials true "Include your credentials"

// üñ•Ô∏è Server-Side CORS Fix (Express.js)
const express = require('express');
const cors = require('cors');
const app = express();
// Method 1: Allow everyone (Development only)
app.use(cors()); // Access-Control-Allow-Origin: *
// Method 2: Allow specific origin (Production recommended)
app.use(cors({
origin: ['http://localhost:3000', 'https://fullstackgada.netlify.app'],
credentials: true // If you need to allow cookies
}));
// Method 3: Manual header setting
app.use((req, res, next) => {
res.header('Access-Control-Allow-Origin', '*');
res.header('Access-Control-Allow-Methods', 'GET,POST,PUT,DELETE');
res.header('Access-Control-Allow-Headers', 'Content-Type,Authorization');
next();
});
app.get('/api/data', (req, res) => {
res.json({ message: 'CORS fixed! Data retrieved successfully!' });
});
