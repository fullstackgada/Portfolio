# üßµ Python Multithreading Example
import threading
import time
lock = threading.Lock() # Traffic light control
def traffic_simulation(car_name, delay):
print(f"{car_name} approaching highway üöó")
with lock: # Only one car can use shared resource
print(f"{car_name} acquired traffic light üö¶")
time.sleep(delay) # Time to cross intersection
print(f"{car_name} released traffic light ‚úÖ")
# Create multiple threads (cars)
cars = [("Car-1", 2), ("Car-2", 1), ("Car-3", 3)]
threads = []
for car_name, delay in cars:
t = threading.Thread(target=traffic_simulation, args=(car_name, delay))
threads.append(t)
t.start()
# Wait for all threads to complete
for t in threads:
t.join()

# ‚öôÔ∏è Python Multiprocessing Example
import multiprocessing as mp
import time
import os
def highway_simulation(car_name, delay):
process_id = os.getpid()
print(f"{car_name} on dedicated highway (PID: {process_id}) üõ£Ô∏è")
# Each process has its own memory space
time.sleep(delay)
print(f"{car_name} reached destination ‚úÖ")
return f"{car_name} completed journey"
if __name__ == "__main__":
cars = [("Car-1", 2), ("Car-2", 1), ("Car-3", 3)]
processes = []
# Create separate processes
for car_name, delay in cars:
p = mp.Process(target=highway_simulation, args=(car_name, delay))
processes.append(p)
p.start()
# Wait for all processes to complete
for p in processes:
p.join()

# üìä Performance Comparison: CPU-bound vs I/O-bound
import threading
import multiprocessing as mp
import time
# CPU-intensive task (computation)
def cpu_bound_task(n):
count = 0
for i in range(n * 1000000):
count += i * i
return count
# I/O-intensive task (waiting)
def io_bound_task(duration):
time.sleep(duration) # Simulate network request
return f"IO task completed in {duration}s"
# Test with different approaches
def benchmark_threading(task, args_list):
start_time = time.time()
threads = []
for args in args_list:
t = threading.Thread(target=task, args=(args,))
threads.append(t)
t.start()
for t in threads:
t.join()
return time.time() - start_time

üîç Complete Comparison: Multithreading vs Multiprocessing

Aspect üßµ Multithreading ‚öôÔ∏è Multiprocessing Best Use Case
Memory Shared memory space Separate memory per process Threading for shared data, Processing for isolation
Communication Direct variable access IPC (pipes, queues, shared memory) Threading for simple communication
Resource Overhead Low (lightweight threads) High (full process creation) Threading for resource efficiency
Parallelism Concurrency only (GIL limitation) True parallelism Processing for CPU-intensive tasks
Failure Impact One thread crash can affect others Process crashes are isolated Processing for fault tolerance
Debugging More complex (shared state) Easier (isolated processes) Processing for debugging ease
Scalability Limited by GIL Scales with CPU cores Processing for multi-core systems
